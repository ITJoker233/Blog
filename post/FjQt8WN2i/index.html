def createTimesatamp(t=32):
    strs = "ABCDEFGHJKMNPQRSTWXYZabcdefhijkmnprstwxyz2345678"
    t = 32

    lens = len(strs)

    img_b64decode= base64.b64decode(image)

def getAtoken(data):
    result = re.match( r'atoken=(.*?);', data, re.M|re.I)
    if result:
        return result.group(1)
    else:
        return 'error'
